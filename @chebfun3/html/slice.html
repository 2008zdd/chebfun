function varargout = slice( f, varargin )

%SLICE  Plots slices of a CHEBFUN3.
%
%   SLICE(F) if F is a CHEBFUN3 then this contains three contour plots of F
%   at slices corresponding to a rough approximation of the max of F.
%
%   SLICE(F,sx,sy,sz) if F is a CHEBFUN3 then this contains three contour
%   plots of F at slices mentioned in sx, sy and sz.

holdState = ishold;
FS = 'fontsize'; FC = 'facecolor'; fs = 18;
dom = f.domain;
[xx, yy, zz] = meshgrid(linspace(dom(1),dom(2),40),linspace(dom(3),dom(4),40),linspace(dom(5),dom(6),40));
%[xx, yy, zz] = meshgrid(dom(1):0.05:dom(2),dom(3):0.05:dom(4),dom(5):0.05:dom(6));
%[xx, yy, zz] = ndgrid(dom(1):0.05:dom(2),dom(3):0.05:dom(4),dom(5):0.05:dom(6));
%[xx, yy, zz] = meshgrid(dom(1):0.025:dom(2),dom(3):0.025:dom(4),dom(5):0.025:dom(6));
v = feval(f,xx,yy,zz);

if ( nargin<2 )
    % Slices are not specified. Choose three yourself.
    [col,row,tube] = ind2sub(size(v),find(v == max(v(:))));
    col = col(end); row = row(end); tube = tube(end);
    xslice = xx(col,row,tube); yslice = yy(col,row,tube); zslice = zz(col,row,tube);
    slice(xx,yy,zz,v,xslice,yslice,zslice);
    str = sprintf('Three slices: x = %2.1g, y = %2.1g, z = %2.1g', xslice, yslice, zslice);
    title(str, FS, fs)
    %title(['Three slices: x = %3.2g' num2str(xslice) ', y = %3.2g' num2str(yslice) ', z = %3.2g' num2str(zslice)], FS, fs)

elseif ( nargin == 4 )
    % Slices are specified.
    %slices = cell2mat(varargin);
    %xslice = slices(1); yslice = slices(2); zslice = slices(3);
    xslice = cell2mat(varargin(1)); yslice = cell2mat(varargin(2)); zslice = cell2mat(varargin(3));
    slice(xx,yy,zz,v,xslice,yslice,zslice);
end

    shading interp
    colorbar,
    set(gca,'xticklabel',{[]})
    set(gca,'yticklabel',{[]})
    set(gca,'zticklabel',{[]})
    xlabel('x',FS, fs); ylabel('y',FS, fs); zlabel('z',FS, fs);
    %title(['f = ' num2str(isoval1)], FS, fs);
     axis tight

if ( ~holdState )
    hold off
end

end % End of function
